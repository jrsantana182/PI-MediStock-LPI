    <script>
        // --- Estado da Aplicação ---
        let inventory = JSON.parse(localStorage.getItem('med_inventory')) || [];
        let movements = JSON.parse(localStorage.getItem('med_movements')) || [];
        let currentUser = null;

        // --- Sistema de Login (Segurança) ---
        const loginForm = document.getElementById('login-form');
        loginForm.addEventListener('submit', (e) => {
            e.preventDefault();
            const user = document.getElementById('username').value;
            const pass = document.getElementById('password').value;

            if (user === 'admin' && pass === '1234') {
                currentUser = { name: 'Administrador LPI', role: 'admin' };
                document.getElementById('login-screen').classList.add('hidden');
                document.getElementById('main-app').classList.remove('hidden');
                initApp();
            } else {
                document.getElementById('login-error').classList.remove('hidden');
            }
        });

        function logout() {
            currentUser = null;
            location.reload();
        }

        // --- Navegação ---
        function showSection(sectionId) {
            document.querySelectorAll('section').forEach(s => s.classList.add('hidden-section'));
            document.getElementById(`section-${sectionId}`).classList.remove('hidden-section');
            
            document.querySelectorAll('.sidebar-link').forEach(l => l.classList.remove('active'));
            event.currentTarget.classList.add('active');

            const titles = { 'dashboard': 'Painel de Controle', 'inventory': 'Cadastro e Estoque', 'movements': 'Registrar Movimentação', 'reports': 'Relatórios e Análises' };
            document.getElementById('current-section-title').innerText = titles[sectionId];
            
            renderAll();
        }

        // --- Core: Cadastro e Dados ---
        const productForm = document.getElementById('product-form');
        productForm.addEventListener('submit', (e) => {
            e.preventDefault();
            const newItem = {
                id: Date.now(),
                name: document.getElementById('p-name').value,
                category: document.getElementById('p-cat').value,
                batch: document.getElementById('p-batch').value,
                quantity: parseInt(document.getElementById('p-qty').value),
                minStock: parseInt(document.getElementById('p-min').value),
                expiry: document.getElementById('p-expiry').value,
                createdAt: new Date().toISOString()
            };

            inventory.push(newItem);
            saveAndRefresh();
            productForm.reset();
        });

        // --- Core: Movimentação (Entrada/Saída) ---
        const movementForm = document.getElementById('movement-form');
        movementForm.addEventListener('submit', (e) => {
            e.preventDefault();
            const prodId = parseInt(document.getElementById('mov-id').value);
            const type = document.getElementById('mov-type').value;
            const qty = parseInt(document.getElementById('mov-qty').value);
            const note = document.getElementById('mov-note').value;

            const product = inventory.find(p => p.id === prodId);
            if (!product) return;

            if (type === 'exit' && product.quantity < qty) {
                showToast("Estoque insuficiente para essa saída!", "bg-red-500");
                return;
            }

            // Atualização automática do saldo
            if (type === 'entry') product.quantity += qty;
            else product.quantity -= qty;

            movements.unshift({
                id: Date.now(),
                productId: prodId,
                productName: product.name,
                type: type,
                quantity: qty,
                note: note,
                date: new Date().toLocaleString()
            });

            saveAndRefresh();
            movementForm.reset();
            showToast("Movimentação registrada com sucesso!", "bg-green-500");
        });

        // --- Renderização Dinâmica ---
        function renderInventory() {
            const list = document.getElementById('inventory-list');
            list.innerHTML = '';
            inventory.forEach(item => {
                const isExpired = new Date(item.expiry) < new Date();
                list.innerHTML += `
                    <tr class="border-b hover:bg-gray-50">
                        <td class="px-6 py-4">
                            <div class="font-medium text-gray-800">${item.name}</div>
                            <div class="text-xs text-gray-500">${item.category}</div>
                        </td>
                        <td class="px-6 py-4 text-sm">${item.batch}</td>
                        <td class="px-6 py-4 text-sm ${isExpired ? 'text-red-600 font-bold' : ''}">${formatDate(item.expiry)}</td>
                        <td class="px-6 py-4">
                            <span class="px-2 py-1 rounded text-xs font-bold ${item.quantity <= item.minStock ? 'bg-orange-100 text-orange-600' : 'bg-green-100 text-green-600'}">
                                ${item.quantity} un
                            </span>
                        </td>
                        <td class="px-6 py-4">
                            <button onclick="deleteItem(${item.id})" class="text-red-500 hover:text-red-700">
                                <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"></path></svg>
                            </button>
                        </td>
                    </tr>
                `;
            });

            // Atualiza select de movimentos
            const select = document.getElementById('mov-id');
            const currentVal = select.value;
            select.innerHTML = '<option value="">Selecione um item...</option>';
            inventory.forEach(item => {
                select.innerHTML += `<option value="${item.id}">${item.name} (Saldo: ${item.quantity})</option>`;
            });
            select.value = currentVal;
        }

        function renderDashboard() {
            const total = inventory.length;
            const low = inventory.filter(i => i.quantity <= i.minStock).length;
            const now = new Date();
            const nextMonth = new Date();
            nextMonth.setMonth(now.getMonth() + 1);

            const expired = inventory.filter(i => new Date(i.expiry) < now).length;
            const near = inventory.filter(i => {
                const d = new Date(i.expiry);
                return d > now && d < nextMonth;
            }).length;

            document.getElementById('stat-total').innerText = total;
            document.getElementById('stat-low').innerText = low;
            document.getElementById('stat-expired').innerText = expired;
            document.getElementById('stat-near-expiry').innerText = near;

            // Alertas Inteligentes
            const alertsArea = document.getElementById('active-alerts');
            alertsArea.innerHTML = '';
            
            inventory.forEach(i => {
                const d = new Date(i.expiry);
                if (d < now) {
                    addAlert(alertsArea, `O item <b>${i.name}</b> (Lote ${i.batch}) está VENCIDO!`, 'bg-red-100 text-red-700 border-red-200');
                } else if (d < nextMonth) {
                    addAlert(alertsArea, `O item <b>${i.name}</b> vence em breve (${formatDate(i.expiry)})`, 'bg-blue-100 text-blue-700 border-blue-200');
                }
                
                if (i.quantity <= i.minStock) {
                    addAlert(alertsArea, `Estoque baixo para <b>${i.name}</b> (${i.quantity} un disponível)`, 'bg-orange-100 text-orange-700 border-orange-200');
                }
            });

            document.getElementById('alert-dot').classList.toggle('hidden', (low + expired + near) === 0);

            // Consulta Rápida
            renderQuickSearch(inventory);
        }

        function renderReports() {
            const criticalList = document.getElementById('report-critical');
            criticalList.innerHTML = '';
            
            const generalList = document.getElementById('report-general-body');
            generalList.innerHTML = '';

            inventory.forEach(i => {
                const isLow = i.quantity <= i.minStock;
                const isExpired = new Date(i.expiry) < new Date();
                const status = isExpired ? 'Vencido' : (isLow ? 'Estoque Crítico' : 'Normal');
                const statusColor = isExpired ? 'text-red-600 font-bold' : (isLow ? 'text-orange-600' : 'text-green-600');

                // Tabela Geral
                generalList.innerHTML += `
                    <tr class="border-b">
                        <td class="px-4 py-2">${i.name}</td>
                        <td class="px-4 py-2 text-xs">${i.batch}</td>
                        <td class="px-4 py-2 text-xs">${formatDate(i.expiry)}</td>
                        <td class="px-4 py-2">${i.minStock}</td>
                        <td class="px-4 py-2 font-bold">${i.quantity}</td>
                        <td class="px-4 py-2 ${statusColor}">${status}</td>
                    </tr>
                `;

                // Itens Críticos
                if (isLow || isExpired) {
                    criticalList.innerHTML += `
                        <div class="flex justify-between items-center bg-gray-50 p-3 rounded border">
                            <div>
                                <p class="font-bold text-sm text-gray-800">${i.name}</p>
                                <p class="text-xs text-gray-500">Motivo: ${status}</p>
                            </div>
                            <div class="text-right">
                                <p class="text-xs font-medium">Saldo: ${i.quantity}</p>
                            </div>
                        </div>
                    `;
                }
            });

            // Histórico
            const historyLog = document.getElementById('history-log');
            historyLog.innerHTML = '';
            movements.forEach(m => {
                historyLog.innerHTML += `
                    <div class="flex items-center text-xs p-2 border-b">
                        <span class="w-24 text-gray-400">${m.date.split(',')[0]}</span>
                        <span class="w-16 font-bold ${m.type === 'entry' ? 'text-green-600' : 'text-red-600'} uppercase">${m.type === 'entry' ? 'Entrada' : 'Saída'}</span>
                        <span class="flex-1"><b>${m.quantity} un</b> de ${m.productName}</span>
                    </div>
                `;
            });
        }

        // --- Helpers ---
        function addAlert(container, msg, classes) {
            container.innerHTML += `
                <div class="alert-pulse p-4 rounded-lg border flex items-center ${classes}">
                    <svg class="w-5 h-5 mr-3" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M12 9v2m0 4h.01m-6.938 4h13.856c1.54 0 2.502-1.667 1.732-3L13.732 4c-.77-1.333-2.694-1.333-3.464 0L3.34 16c-.77 1.333.192 3 1.732 3z"></path></svg>
                    <span class="text-sm">${msg}</span>
                </div>
            `;
        }

        function renderQuickSearch(data) {
            const body = document.getElementById('quick-search-body');
            body.innerHTML = '';
            data.forEach(i => {
                const isLow = i.quantity <= i.minStock;
                body.innerHTML += `
                    <tr class="border-b">
                        <td class="px-4 py-2 font-medium">${i.name}</td>
                        <td class="px-4 py-2 text-gray-500">${i.category}</td>
                        <td class="px-4 py-2 font-bold">${i.quantity} un</td>
                        <td class="px-4 py-2">
                            <span class="w-3 h-3 inline-block rounded-full ${isLow ? 'bg-orange-500' : 'bg-green-500'}"></span>
                        </td>
                    </tr>
                `;
            });
        }

        function filterQuickSearch() {
            const term = document.getElementById('quick-search').value.toLowerCase();
            const filtered = inventory.filter(i => 
                i.name.toLowerCase().includes(term) || 
                i.category.toLowerCase().includes(term)
            );
            renderQuickSearch(filtered);
        }

        function deleteItem(id) {
            if (confirm("Deseja realmente remover este medicamento do cadastro?")) {
                inventory = inventory.filter(i => i.id !== id);
                saveAndRefresh();
            }
        }

        function saveAndRefresh() {
            localStorage.setItem('med_inventory', JSON.stringify(inventory));
            localStorage.setItem('med_movements', JSON.stringify(movements));
            renderAll();
        }

        function renderAll() {
            renderInventory();
            renderDashboard();
            renderReports();
        }

        function formatDate(dateStr) {
            if (!dateStr) return "-";
            const [y, m, d] = dateStr.split('-');
            return `${d}/${m}/${y}`;
        }

        function showToast(msg, colorClass) {
            const toast = document.createElement('div');
            toast.className = `fixed bottom-5 right-5 text-white px-6 py-3 rounded-lg shadow-lg z-[100] transition-opacity duration-500 ${colorClass}`;
            toast.innerText = msg;
            document.body.appendChild(toast);
            setTimeout(() => {
                toast.style.opacity = '0';
                setTimeout(() => toast.remove(), 500);
            }, 3000);
        }

        function initApp() {
            renderAll();
        }
